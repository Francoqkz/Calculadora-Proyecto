    private void IgualActionPerformed(java.awt.event.ActionEvent evt) {                                      
    SegundoNumero = Double.parseDouble(pantalla.getText());
    // y por ultimo le digo como proceder a la calculadora para que ejecute las operaciones correspondientes
    if(Operador.equals("+")){
        Resultado = PrimerNumero + SegundoNumero;
        pantalla.setText(Double.toString(Resultado));  
    }else if (Operador.equals("-")){
        Resultado = PrimerNumero - SegundoNumero;
        pantalla.setText(Double.toString(Resultado));  
        
    }else if (Operador.equals("*")){
        Resultado = PrimerNumero * SegundoNumero;
        pantalla.setText(Double.toString(Resultado));
        
    }else if (Operador.equals("/")){
        Resultado = PrimerNumero / SegundoNumero;
        pantalla.setText(Double.toString(Resultado));  
    } 
    }
